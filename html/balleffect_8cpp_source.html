105 void UpdateBalleffect(void)
106 {
107  for (int i = 0; i < BALLEFFECT_NUM_BALLEFFECTS; i++)
108  {
109  if (balleffectWk[i].use)
110  {
111  if (balleffectWk[i].isEnding) {
112  balleffectWk[i].use = FALSE;
113  continue;
114  }
115 
116  //エミットが有効であればエフェクト作成処理を行う
117  if (balleffectWk[i].isRemoveOnFinish == FALSE)
118  {
119 
120  //エフェクト作成レートの増加処理
121  if (balleffectWk[i].balleffectCount < BALLEFFECT_NUM_PARTS)
122  balleffectWk[i].emitCounter++;
123 
124  //バッファに空きがあり、追加タイミングが来ていれば新たなエフェクトを追加する
125  while ((balleffectWk[i].balleffectCount < BALLEFFECT_NUM_PARTS) && (balleffectWk[i].emitCounter > EMISSION_RATE))
126  {
127  //全体追加フラグがONであれば空き領域全てに新たなエフェクトを追加する
128  if (EMISSION_FULL)
129  balleffectWk[i].balleffectCount = BALLEFFECT_NUM_PARTS;
130  else
131  balleffectWk[i].balleffectCount++;
132 
133  //エフェクト作成レートの初期化
134  balleffectWk[i].emitCounter = 0;
135  }
136 
137  balleffectWk[i].elapsed++;
138 
139  //時間超過
140  if ((balleffectWk[i].duration != -1) && (balleffectWk[i].duration < balleffectWk[i].elapsed)) {
141  balleffectWk[i].isRemoveOnFinish = TRUE;
142  }
143  }
144 
145  int balleffectIndex = 0;
146 
147  //エフェクトの更新処理
148  while (balleffectIndex < balleffectWk[i].balleffectCount)
149  {
150  if (balleffectWk[i].pParticle[balleffectIndex].liveTime > 0)
151  {
152  //生存フレーム減少
153  balleffectWk[i].pParticle[balleffectIndex].liveTime--;
154 
155  //アニメパターン進行
156  if (++balleffectWk[i].pParticle[balleffectIndex].CountAnim > BALLEFFECT_TIME_ANIMATION) {
157  //アニメパターンが最大値に達した場合でも終了
158  if (++balleffectWk[i].pParticle[balleffectIndex].PatternAnim >= (BALLEFFECT_ANIM_PATTERN_NUM - 1))
159  {
160  balleffectWk[i].pParticle[balleffectIndex].PatternAnim = BALLEFFECT_ANIM_PATTERN_NUM - 1;
161  balleffectWk[i].pParticle[balleffectIndex].liveTime = 0;
162  }
163 
164  balleffectWk[i].pParticle[balleffectIndex].CountAnim = 0;
165  }
166 
167  SetTextureBalleffect(i, balleffectIndex, balleffectWk[i].pParticle[balleffectIndex].PatternAnim);
168 
169  SetVertexBalleffect(i, balleffectIndex); // 移動後の座標で頂点を設定
170 
171  balleffectIndex++;
172  }
173  else {
174  if (balleffectWk[i].isRemoveOnFinish) {
175  if (balleffectWk[i].pParticle[balleffectIndex].isFinish == 0)
176  {
177  balleffectWk[i].pParticle[balleffectIndex].isFinish = 1;
178  balleffectWk[i].numFinish++;
179  }
180  //終了処理
181  if (balleffectWk[i].pParticle[balleffectIndex].pos.x <= SCREEN_W - SCREEN_W || balleffectWk[i].pParticle[balleffectIndex].pos.x >= SCREEN_W ||
182  balleffectWk[i].pParticle[balleffectIndex].pos.y <= SCREEN_H - SCREEN_H || balleffectWk[i].pParticle[balleffectIndex].pos.y >= SCREEN_H)
183  {
184  balleffectWk[i].isEnding = TRUE;
185  break;
186  }
187  if (balleffectWk[i].numFinish == balleffectWk[i].balleffectCount)
188  {
189  balleffectWk[i].isEnding = TRUE;
190  break;
191  }
192  balleffectIndex++;
193  }
194  else {
195  //バッファを初期化する
196  ResetParticle(i, balleffectIndex);
197 
198  //末尾でなければインデックスを詰める
199  if (balleffectIndex != (balleffectWk[i].balleffectCount - 1))
200  {
201  balleffectWk[i].pParticle[balleffectIndex] = balleffectWk[i].pParticle[balleffectWk[i].balleffectCount - 1];
202  ResetParticle(i, (balleffectWk[i].balleffectCount - 1));
203  }
