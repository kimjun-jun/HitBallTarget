106 void UpdateBalleffect(void)
107 {
108  for (int i = 0; i < BALLEFFECT_NUM_BALLEFFECTS; i++)
109  {
110  if (balleffectWk[i].use)
111  {
112  if (balleffectWk[i].isEnding) {
113  balleffectWk[i].use = FALSE;
114  continue;
115  }
116 
117  //エミットが有効であればエフェクト作成処理を行う
118  if (balleffectWk[i].isRemoveOnFinish == FALSE)
119  {
120 
121  //エフェクト作成レートの増加処理
122  if (balleffectWk[i].balleffectCount < BALLEFFECT_NUM_PARTS)
123  balleffectWk[i].emitCounter++;
124 
125  //バッファに空きがあり、追加タイミングが来ていれば新たなエフェクトを追加する
126  while ((balleffectWk[i].balleffectCount < BALLEFFECT_NUM_PARTS) && (balleffectWk[i].emitCounter > EMISSION_RATE))
127  {
128  //全体追加フラグがONであれば空き領域全てに新たなエフェクトを追加する
129  if (EMISSION_FULL)
130  balleffectWk[i].balleffectCount = BALLEFFECT_NUM_PARTS;
131  else
132  balleffectWk[i].balleffectCount++;
133 
134  //エフェクト作成レートの初期化
135  balleffectWk[i].emitCounter = 0;
136  }
137 
138  balleffectWk[i].elapsed++;
139 
140  //時間超過
141  if ((balleffectWk[i].duration != -1) && (balleffectWk[i].duration < balleffectWk[i].elapsed)) {
142  balleffectWk[i].isRemoveOnFinish = TRUE;
143  }
144  }
145 
146  int balleffectIndex = 0;
147 
148  //エフェクトの更新処理
149  while (balleffectIndex < balleffectWk[i].balleffectCount)
150  {
151  if (balleffectWk[i].pParticle[balleffectIndex].liveTime > 0)
152  {
153  //生存フレーム減少
154  balleffectWk[i].pParticle[balleffectIndex].liveTime--;
155 
156  //アニメパターン進行
157  if (++balleffectWk[i].pParticle[balleffectIndex].CountAnim > BALLEFFECT_TIME_ANIMATION) {
158  //アニメパターンが最大値に達した場合でも終了
159  if (++balleffectWk[i].pParticle[balleffectIndex].PatternAnim >= (BALLEFFECT_ANIM_PATTERN_NUM - 1))
160  {
161  balleffectWk[i].pParticle[balleffectIndex].PatternAnim = BALLEFFECT_ANIM_PATTERN_NUM - 1;
162  balleffectWk[i].pParticle[balleffectIndex].liveTime = 0;
163  }
164 
165  balleffectWk[i].pParticle[balleffectIndex].CountAnim = 0;
166  }
167 
168  SetTextureBalleffect(i, balleffectIndex, balleffectWk[i].pParticle[balleffectIndex].PatternAnim);
169 
170  SetVertexBalleffect(i, balleffectIndex); // 移動後の座標で頂点を設定
171 
172  balleffectIndex++;
173  }
174  else {
175  if (balleffectWk[i].isRemoveOnFinish) {
176  if (balleffectWk[i].pParticle[balleffectIndex].isFinish == 0)
177  {
178  balleffectWk[i].pParticle[balleffectIndex].isFinish = 1;
179  balleffectWk[i].numFinish++;
180  }
181  //終了処理
182  if (balleffectWk[i].pParticle[balleffectIndex].pos.x <= SCREEN_W - SCREEN_W || balleffectWk[i].pParticle[balleffectIndex].pos.x >= SCREEN_W ||
183  balleffectWk[i].pParticle[balleffectIndex].pos.y <= SCREEN_H - SCREEN_H || balleffectWk[i].pParticle[balleffectIndex].pos.y >= SCREEN_H)
184  {
185  balleffectWk[i].isEnding = TRUE;
186  break;
187  }
188  if (balleffectWk[i].numFinish == balleffectWk[i].balleffectCount)
189  {
190  balleffectWk[i].isEnding = TRUE;
191  break;
192  }
193  balleffectIndex++;
194  }
195  else {
196  //バッファを初期化する
197  ResetParticle(i, balleffectIndex);
198 
199  //末尾でなければインデックスを詰める
200  if (balleffectIndex != (balleffectWk[i].balleffectCount - 1))
201  {
202  balleffectWk[i].pParticle[balleffectIndex] = balleffectWk[i].pParticle[balleffectWk[i].balleffectCount - 1];
203  ResetParticle(i, (balleffectWk[i].balleffectCount - 1));
204  }
205  balleffectWk[i].balleffectCount--;
206  }
207  }
208  }
209  }
210  }
