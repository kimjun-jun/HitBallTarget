42  //「推測」→もし2次元ならD3DXVECTOR2を使う。今回player->posとenemy->posを3次元で定義しているためD3DXVECTOR3を使ってplayer_balletを3次元定義している
43  D3DXVECTOR3 distance_pos = pos1 - pos2;
44 
45  //式2 プレイヤーサークル(ボディ)の半径
46  //playerのpos(D3DXVECTOR3なので3次元)のx座標から当たり判定用テクスチャサイズ(プレイヤー半径)を差し引いた値をplayer_r(プレイヤー半径)に代入
47  float wr1 = r1;
48  //式2 エネミーサークル(ボディ)の半径
49  float wr2 = r2;
50 
51  //D3DXVec2LengthSq()関数でdistance_pos(3次元)を2乗している
52  //D3DXVec2Length()関数だとdistance_pos(3次元)を2乗したものに√をつける
53  float dista = D3DXVec3LengthSq(&distance_pos);
54 
55  float wr3 = (wr1 + wr2)*(wr1 + wr2);
56 
57  //当たり判定
58  if (dista <= wr3)
59  {
60  //当たっている
61  return true;
62  }
63  return false;
64 }
