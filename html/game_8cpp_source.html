271 void Ball(void)
272 {
273  BALL *b = GetBall();
274  BALLCURSOR *bc = GetBallcursor();
275  BALLSHADOW *bs = GetBallshadow();
276  PLAYER_BAT *pb = GetPlayerBat();
277  //ランダム投球
278  //左右にホーミング。old移動量を控えといて、yが基準点になったらそれ以降はold移動量で動かす
279  if (b->OorD == true)
280  {
281  bc->movecount++;
282  if (bc->movecount > 100) bc->movecount = 0;
283  if (bc->movecount % TIME_BALLCURSOL_MOVECOUNT_MID == 0)
284  {
285  switch (rand() % 6)
286  {
287  case DIRECTION_LEFT:
288  bc->direction = DIRECTION_LEFT;
289  break;
290  case DIRECTION_RIGHT:
291  bc->direction = DIRECTION_RIGHT;
292  break;
293  case DIRECTION_UP:
294  bc->direction = DIRECTION_UP;
295  break;
296  case DIRECTION_DOWN:
297  bc->direction = DIRECTION_DOWN;
298  break;
299  default:
300  bc->direction = 100;//移動しない
301  break;
302  }
303  }
304  //bcがランダムで上下左右に移動
305  if (bc->direction == DIRECTION_LEFT) bc->pos.x -= BALLCURSOR_MOTION_SPEED;
306  else if (bc->direction == DIRECTION_RIGHT) bc->pos.x += BALLCURSOR_MOTION_SPEED;
307  else if (bc->direction == DIRECTION_UP)
308  {
309  b->changeValTypeA += BALL_CHENGEVALTYPEA_SIZE;
310  b->changeVal += BALL_CHENGEVAL_SIZE;
311  bc->pos.y -= BALLCURSOR_MOTION_SPEED;
312  }
313  else if (bc->direction == DIRECTION_DOWN)
314  {
315  b->changeValTypeA -= BALL_CHENGEVALTYPEA_SIZE;
316  b->changeVal -= BALL_CHENGEVAL_SIZE;
317  bc->pos.y += BALLCURSOR_MOTION_SPEED;
318  }
319  //bがbcいくまでy座標上がる
320  if (b->pos.y > bc->pos.y)
321  {
322  b->pos += b->oldmove;
323  bs->pos += bs->oldmove;
324  b->pos.y += BALL_MOTION_SPEED_FAST;
325  bs->pos.y += BALL_MOTION_SPEED_FAST;
326  }
327  //bがbcにホーミング処理
328  else if (b->pos.x + 10 < bc->pos.x || b->pos.x > bc->pos.x + 10)
329  {
330  D3DXVECTOR3 pos = bc->pos - b->pos;
331  float rot = atan2f(pos.y, pos.x);
332  b->pos.x += cosf(rot) * BALL_MOTION_SPEED_FAST;
333  b->pos.y += sinf(rot) * BALL_MOTION_SPEED_FAST;
334  bs->pos.x += cosf(rot) * BALL_MOTION_SPEED_FAST;
335  bs->pos.y += sinf(rot) * BALL_MOTION_SPEED_FAST;
336  if (b->pos.y > bc->pos.y - 5)
337  {
338  b->oldmove.x = cosf(rot) * BALL_MOTION_SPEED_FAST;
339  b->oldmove.y = sinf(rot) * BALL_MOTION_SPEED_FAST;
340  bs->oldmove.x = cosf(rot) * BALL_MOTION_SPEED_FAST;
341  bs->oldmove.y = sinf(rot) * BALL_MOTION_SPEED_FAST;
342  }
343  }
344  //ホーミング後の誤差が出てもここで無理やりyを上げる
345  else
346  {
347  b->pos.y += BALL_MOTION_SPEED_FAST;
348  bs->pos.y += BALL_MOTION_SPEED_FAST;
349  }
350 
351  //バッティング可能でボールカーソルを赤色に変更。ここをZ軸で判定するかY軸で判定するか
352  if (b->pos.y >= bc->pos.y - BALL_TIMING_PULL_KARABURI)
353  {
354  bc->vertexWk[0].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
355  bc->vertexWk[1].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
356  bc->vertexWk[2].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
357  bc->vertexWk[3].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
358  }
359  else
360  {
361  bc->vertexWk[0].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
362  bc->vertexWk[1].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
363  bc->vertexWk[2].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
364  bc->vertexWk[3].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
365  }
366  b[1].pos = b[2].pos = b[0].pos;
367  b->changeValTypeA += BALL_CHENGEVALTYPEA_SIZE;
368  if (b->changeValTypeA >= BALL_CHENGEVALTYPEA_MIN) b->changeValTypeA -= BALL_CHENGEVALTYPEA_SIZE;
369  else if (b->changeValTypeA <= BALL_CHENGEVALTYPEA_MAX) b->changeValTypeA += BALL_CHENGEVALTYPEA_SIZE;
370  b->changeVal += BALL_CHENGEVAL_SIZE;
371  if (b->changeVal >= BALL_CHENGEVAL_MIN) b->changeVal -= BALL_CHENGEVAL_SIZE;
372  else if (b->changeVal <= BALL_CHENGEVAL_MAX) b->changeVal += BALL_CHENGEVAL_SIZE;
373 
374  }
375 }
