270 void Ball(void)
271 {
272  BALL *b = GetBall();
273  BALLCURSOR *bc = GetBallcursor();
274  BALLSHADOW *bs = GetBallshadow();
275  PLAYER_BAT *pb = GetPlayerBat();
276  //ランダム投球
277  //左右にホーミング。old移動量を控えといて、yが基準点になったらそれ以降はold移動量で動かす
278  if (b->OorD == true)
279  {
280  bc->movecount++;
281  if (bc->movecount > 100) bc->movecount = 0;
282  if (bc->movecount % TIME_BALLCURSOL_MOVECOUNT_MID == 0)
283  {
284  switch (rand() % 6)
285  {
286  case DIRECTION_LEFT:
287  bc->direction = DIRECTION_LEFT;
288  break;
289  case DIRECTION_RIGHT:
290  bc->direction = DIRECTION_RIGHT;
291  break;
292  case DIRECTION_UP:
293  bc->direction = DIRECTION_UP;
294  break;
295  case DIRECTION_DOWN:
296  bc->direction = DIRECTION_DOWN;
297  break;
298  default:
299  bc->direction = 100;//移動しない
300  break;
301  }
302  }
303  //bcがランダムで上下左右に移動
304  if (bc->direction == DIRECTION_LEFT) bc->pos.x -= BALLCURSOR_MOTION_SPEED;
305  else if (bc->direction == DIRECTION_RIGHT) bc->pos.x += BALLCURSOR_MOTION_SPEED;
306  else if (bc->direction == DIRECTION_UP)
307  {
308  b->changeValTypeA += BALL_CHENGEVALTYPEA_SIZE;
309  b->changeVal += BALL_CHENGEVAL_SIZE;
310  bc->pos.y -= BALLCURSOR_MOTION_SPEED;
311  }
312  else if (bc->direction == DIRECTION_DOWN)
313  {
314  b->changeValTypeA -= BALL_CHENGEVALTYPEA_SIZE;
315  b->changeVal -= BALL_CHENGEVAL_SIZE;
316  bc->pos.y += BALLCURSOR_MOTION_SPEED;
317  }
318  //bがbcいくまでy座標上がる
319  if (b->pos.y > bc->pos.y)
320  {
321  b->pos += b->oldmove;
322  bs->pos += bs->oldmove;
323  b->pos.y += BALL_MOTION_SPEED_FAST;
324  bs->pos.y += BALL_MOTION_SPEED_FAST;
325  }
326  //bがbcにホーミング処理
327  else if (b->pos.x + 10 < bc->pos.x || b->pos.x > bc->pos.x + 10)
328  {
329  D3DXVECTOR3 pos = bc->pos - b->pos;
330  float rot = atan2f(pos.y, pos.x);
331  b->pos.x += cosf(rot) * BALL_MOTION_SPEED_FAST;
332  b->pos.y += sinf(rot) * BALL_MOTION_SPEED_FAST;
333  bs->pos.x += cosf(rot) * BALL_MOTION_SPEED_FAST;
334  bs->pos.y += sinf(rot) * BALL_MOTION_SPEED_FAST;
335  if (b->pos.y > bc->pos.y - 5)
336  {
337  b->oldmove.x = cosf(rot) * BALL_MOTION_SPEED_FAST;
338  b->oldmove.y = sinf(rot) * BALL_MOTION_SPEED_FAST;
339  bs->oldmove.x = cosf(rot) * BALL_MOTION_SPEED_FAST;
340  bs->oldmove.y = sinf(rot) * BALL_MOTION_SPEED_FAST;
341  }
342  }
343  //ホーミング後の誤差が出てもここで無理やりyを上げる
344  else
345  {
346  b->pos.y += BALL_MOTION_SPEED_FAST;
347  bs->pos.y += BALL_MOTION_SPEED_FAST;
348  }
349 
350  //バッティング可能でボールカーソルを赤色に変更。ここをZ軸で判定するかY軸で判定するか
351  if (b->pos.y >= bc->pos.y - BALL_TIMING_PULL1)
352  {
353  bc->vertexWk[0].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
354  bc->vertexWk[1].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
355  bc->vertexWk[2].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
356  bc->vertexWk[3].diffuse = D3DCOLOR_RGBA(255, 0, 0, 255);
357  }
358  else
359  {
360  bc->vertexWk[0].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
361  bc->vertexWk[1].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
362  bc->vertexWk[2].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
363  bc->vertexWk[3].diffuse = D3DCOLOR_RGBA(0, 0, 0, 100);
364  }
365  b[1].pos = b[2].pos = b[0].pos;
366  b->changeValTypeA += BALL_CHENGEVALTYPEA_SIZE;
367  if (b->changeValTypeA >= BALL_CHENGEVALTYPEA_MIN) b->changeValTypeA -= BALL_CHENGEVALTYPEA_SIZE;
368  else if (b->changeValTypeA <= BALL_CHENGEVALTYPEA_MAX) b->changeValTypeA += BALL_CHENGEVALTYPEA_SIZE;
369  b->changeVal += BALL_CHENGEVAL_SIZE;
370  if (b->changeVal >= BALL_CHENGEVAL_MIN) b->changeVal -= BALL_CHENGEVAL_SIZE;
371  else if (b->changeVal <= BALL_CHENGEVAL_MAX) b->changeVal += BALL_CHENGEVAL_SIZE;
372 
373  }
374 }
