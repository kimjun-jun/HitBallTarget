181 HRESULT UpdateKeyboard(void)
182 {
183  HRESULT hr;
184  BYTE keyStateOld[256];
185 
186  // 前回のデータを保存
187  memcpy(keyStateOld, g_keyState, NUM_KEY_MAX);
188 
189  // デバイスからデータを取得
190  hr = g_pDIDevKeyboard->GetDeviceState(sizeof(g_keyState), g_keyState);
191  if (SUCCEEDED(hr))
192  {
193  for (int cnt = 0; cnt < NUM_KEY_MAX; cnt++)
194  {
195  g_keyStateTrigger[cnt] = (keyStateOld[cnt] ^ g_keyState[cnt]) & g_keyState[cnt];
196  g_keyStateRelease[cnt] = (keyStateOld[cnt] ^ g_keyState[cnt]) & ~g_keyState[cnt];
197  g_keyStateRepeat[cnt] = g_keyStateTrigger[cnt];
198 
199  if (g_keyState[cnt])
200  {
201  g_keyStateRepeatCnt[cnt]++;
202  if (g_keyStateRepeatCnt[cnt] >= 20)
203  {
204  g_keyStateRepeat[cnt] = g_keyState[cnt];
205  }
206  }
207  else
208  {
209  g_keyStateRepeatCnt[cnt] = 0;
210  g_keyStateRepeat[cnt] = 0;
211  }
212  }
213  }
214  else
215  {
216  // キーボードへのアクセス権を取得
217  g_pDIDevKeyboard->Acquire();
218  }
219 
220  return S_OK;
221 }

394 HRESULT InitializePad(void) // パッド初期化
395 {
396  HRESULT result;
397  int i;
398 
399  padCount = 0;
400  // ジョイパッドを探す
401  g_pDInput->EnumDevices(DI8DEVCLASS_GAMECTRL, (LPDIENUMDEVICESCALLBACK)SearchGamePadCallback, NULL, DIEDFL_ATTACHEDONLY);
402  // セットしたコールバック関数が、パッドを発見した数だけ呼ばれる。
403 
404  for (i = 0; i < padCount; i++) {
405  // ジョイスティック用のデータ・フォーマットを設定
406  result = pGamePad[i]->SetDataFormat(&c_dfDIJoystick);
407  if (FAILED(result))
408  return false; // データフォーマットの設定に失敗
409 
410  // モードを設定（フォアグラウンド＆非排他モード）
411 // result = pGamePad[i]->SetCooperativeLevel(hWindow, DISCL_NONEXCLUSIVE | DISCL_FOREGROUND);
412 // if ( FAILED(result) )
413 // return false; // モードの設定に失敗
414 
415  // 軸の値の範囲を設定
416  // X軸、Y軸のそれぞれについて、オブジェクトが報告可能な値の範囲をセットする。
417  // (max-min)は、最大10,000(?)。(max-min)/2が中央値になる。
418  // 差を大きくすれば、アナログ値の細かな動きを捕らえられる。(パッドの性能による)
419  DIPROPRANGE diprg;
420  ZeroMemory(&diprg, sizeof(diprg));
421  diprg.diph.dwSize = sizeof(diprg);
422  diprg.diph.dwHeaderSize = sizeof(diprg.diph);
423  diprg.diph.dwHow = DIPH_BYOFFSET;
424  diprg.lMin = RANGE_MIN;
425  diprg.lMax = RANGE_MAX;
426  // X軸の範囲を設定
427  diprg.diph.dwObj = DIJOFS_X;
428  pGamePad[i]->SetProperty(DIPROP_RANGE, &diprg.diph);
429  // Y軸の範囲を設定
430  diprg.diph.dwObj = DIJOFS_Y;
431  pGamePad[i]->SetProperty(DIPROP_RANGE, &diprg.diph);
432 
433  // 各軸ごとに、無効のゾーン値を設定する。
434  // 無効ゾーンとは、中央からの微少なジョイスティックの動きを無視する範囲のこと。
435  // 指定する値は、10000に対する相対値(2000なら20パーセント)。
436  DIPROPDWORD dipdw;
437  dipdw.diph.dwSize = sizeof(DIPROPDWORD);
438  dipdw.diph.dwHeaderSize = sizeof(dipdw.diph);
439  dipdw.diph.dwHow = DIPH_BYOFFSET;
440  dipdw.dwData = DEADZONE;
441  //X軸の無効ゾーンを設定
442  dipdw.diph.dwObj = DIJOFS_X;
443  pGamePad[i]->SetProperty(DIPROP_DEADZONE, &dipdw.diph);
444  //Y軸の無効ゾーンを設定
445  dipdw.diph.dwObj = DIJOFS_Y;
446  pGamePad[i]->SetProperty(DIPROP_DEADZONE, &dipdw.diph);
447 
448  //ジョイスティック入力制御開始
449  pGamePad[i]->Acquire();
450  }
451 
452  return true;
