313 HRESULT CheckChunk(HANDLE hFile, DWORD format, DWORD *pChunkSize, DWORD *pChunkDataPosition)
314 {
315  HRESULT hr = S_OK;
316  DWORD dwRead;
317  DWORD dwChunkType;
318  DWORD dwChunkDataSize;
319  DWORD dwRIFFDataSize = 0;
320  DWORD dwFileType;
321  DWORD dwBytesRead = 0;
322  DWORD dwOffset = 0;
323 
324  if (SetFilePointer(hFile, 0, NULL, FILE_BEGIN) == INVALID_SET_FILE_POINTER)
325  {// ファイルポインタを先頭に移動
326  return HRESULT_FROM_WIN32(GetLastError());
327  }
328 
329  while (hr == S_OK)
330  {
331  if (ReadFile(hFile, &dwChunkType, sizeof(DWORD), &dwRead, NULL) == 0)
332  {// チャンクの読み込み
333  hr = HRESULT_FROM_WIN32(GetLastError());
334  }
335 
336  if (ReadFile(hFile, &dwChunkDataSize, sizeof(DWORD), &dwRead, NULL) == 0)
337  {// チャンクデータの読み込み
338  hr = HRESULT_FROM_WIN32(GetLastError());
339  }
340 
341  switch (dwChunkType)
342  {
343  case 'FFIR':
344  dwRIFFDataSize = dwChunkDataSize;
345  dwChunkDataSize = 4;
346  if (ReadFile(hFile, &dwFileType, sizeof(DWORD), &dwRead, NULL) == 0)
347  {// ファイルタイプの読み込み
348  hr = HRESULT_FROM_WIN32(GetLastError());
349  }
350  break;
351 
352  default:
353  if (SetFilePointer(hFile, dwChunkDataSize, NULL, FILE_CURRENT) == INVALID_SET_FILE_POINTER)
354  {// ファイルポインタをチャンクデータ分移動
355  return HRESULT_FROM_WIN32(GetLastError());
356  }
357  }
358 
359  dwOffset += sizeof(DWORD) * 2;
360  if (dwChunkType == format)
361  {
362  *pChunkSize = dwChunkDataSize;
363  *pChunkDataPosition = dwOffset;
364 
365  return S_OK;
366  }
367 
368  dwOffset += dwChunkDataSize;
369  if (dwBytesRead >= dwRIFFDataSize)
370  {
371  return S_FALSE;
372  }
373  }
374 
375  return S_OK;
376 }
